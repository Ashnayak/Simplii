## Python installation (Linux)

sudo apt-get update
sudo apt-get upgrade

sudo apt install python3.8-venv

## Flask installation (Mac)
python3 -m venv venv


. venv/bin/activate


pip install Flask


## Python installation (Windows) 
py -3 -m venv venv

venv\Scripts\activate

pip install Flask
